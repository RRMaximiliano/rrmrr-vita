% --- 1. Main Section Container (\cvsection) ---
\newcommand{\cvsection}[2]{%
  \par\vspace{1pt}
  \begin{list}{}{%
      \setlength{\leftmargin}{0.22\textwidth}
      \setlength{\labelwidth}{0pt}
      \setlength{\labelsep}{0pt}
      \setlength{\topsep}{0pt}
      \setlength{\partopsep}{0pt}
      \setlength{\itemsep}{0pt}
      \setlength{\parsep}{0pt}
    }
  \item
    \makebox[0pt][r]{%
      \smash{%
        \begin{minipage}[t]{0.20\textwidth}
          \raggedright\textbf{#1}
        \end{minipage}%
      }%
      \hspace{0.02\textwidth}%
    }%
    \ignorespaces
    #2
  \end{list}
}

% --- 2. Standard Item (\cvitem) ---
\newcommand{\cvitem}[4][\textbf]{%
  \noindent
  {#1{#2}}\strut \hfill {\small #4}%
  \par
  \ifx&#3&%
  \vspace{2pt}
  \else
  \nointerlineskip
  \vspace{1pt}
  {%
    \addtolength{\leftskip}{1em}
    \small
    \noindent #3
    \par
  }%
  \vspace{2pt}
  \fi
}

% --- 3. Conference/Seminar Item (\cvitemconf) ---
\newcommand{\cvitemconf}[2]{%
  \noindent
  \begin{minipage}[t]{0.87\linewidth}
    \raggedright \textnormal{#1}\strut
  \end{minipage}%
  \hfill%
  \begin{minipage}[t]{0.13\linewidth}
    \raggedleft {\small #2}\strut
  \end{minipage}%
  \par
  \nointerlineskip
  \vspace{2pt}
}

% --- 4. Publication/Paper Item (\cvitempaper) ---
\newcommand{\cvitempaper}[1]{%
  \noindent
  \hangindent=2em
  \hangafter=1
  #1
  \par\vspace{2pt}
}
