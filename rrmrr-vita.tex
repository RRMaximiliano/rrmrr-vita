% --- 1. Main Section Container (\cvsection) ---
% Creates a two-column layout:
% Left: Section Title (e.g., "Education")
% Right: Content List (allows page breaks)
% ------------------------------------------------------------------------
\newcommand{\cvsection}[2]{%
  \par\vspace{4pt} 
  \begin{list}{}{%
      \setlength{\leftmargin}{0.26\textwidth} 
      \setlength{\labelwidth}{0pt}
      \setlength{\labelsep}{0pt}
      \setlength{\topsep}{0pt}
      \setlength{\partopsep}{0pt}
      \setlength{\itemsep}{0pt}
      \setlength{\parsep}{0pt}
  }
    \item 
    \makebox[0pt][r]{%
      % \smash hides the vertical height/depth of the title.
      % This prevents long titles from messing up the first item's spacing.
      \smash{%
        \begin{minipage}[t]{0.24\textwidth}
          \raggedright\textbf{#1}
        \end{minipage}%
      }%
      \hspace{0.02\textwidth}%
    }%
    \ignorespaces 
    #2
  \end{list}
}


% --- 2. Standard Item (\cvitem) ---
% Used for: Education, Teaching Experience, Awards
% Args: [Font Style] {Title} {Description} {Date}
% ------------------------------------------------------------------------
\newcommand{\cvitem}[4][\textbf]{%
  \noindent
  {#1{#2}}\strut \hfill {\small #4}% <--- Year is now forced to SMALL
  \par
  
  \ifx&#3&%
    % Case: No Description
    \vspace{4pt} 
  \else
    % Case: Has Description
    \nointerlineskip
    \vspace{2pt} 
    {% 
      \addtolength{\leftskip}{1em} 
      \small % <--- Forces the entire Description block to SMALL
      \noindent #3 
      \par 
    }%
    \vspace{4pt} 
  \fi
}


% --- 3. Conference/Seminar Item (\cvitemconf) ---
% Used for: Lists of talks where text might wrap
% Uses columns so the Date stays top-aligned even if text wraps
% ------------------------------------------------------------------------
\newcommand{\cvitemconf}[2]{%
  \noindent
  % Left Column
  \begin{minipage}[t]{0.87\linewidth}
    \raggedright \textnormal{#1}\strut % \strut ensures consistent line height
  \end{minipage}%
  \hfill%
  % Right Column
  \begin{minipage}[t]{0.13\linewidth}
    \raggedleft {\small #2}\strut 
  \end{minipage}%
  \par
  % THE FIX:
  \nointerlineskip % 1. Ignore variable baseline calculations
  \vspace{4pt}     % 2. Add consistent spacing (Ink-to-Ink)
}

% --- 4. Publication/Paper Item (\cvitempaper) ---
% Used for: Bibliography style entries
% Why: Creates a "Hanging Indent" (French indent) for references
% ------------------------------------------------------------------------
\newcommand{\cvitempaper}[1]{%
  \noindent % Forces alignment
  \hangindent=2em % Indent subsequent lines
  \hangafter=1    % Start indent after the first line
  #1
  \par\vspace{4pt} % Space after item
}
